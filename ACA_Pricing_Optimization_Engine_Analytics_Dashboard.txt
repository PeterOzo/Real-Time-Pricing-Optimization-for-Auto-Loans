        function trainedPricingEngine(loanAmnt, annualInc, dti, intRate, gradeNum) {
            // Grade-based risk mapping from Lending Club training
            const gradeRisks = [0.04, 0.08, 0.14, 0.22, 0.35, 0.50, 0.70];
            let risk = gradeRisks[Math.min(gradeNum - 1, 6)];
            
            // Adjustments based on Lending Club training logic
            if (annualInc < 40000) risk *= 1.3;
            if (dti > 25) risk *= 1.2;
            
            // Rate calculation from trained model
            const baseRate = 14.0;
            let rate = baseRate + (risk - 0.12) * 10;
            rate = Math.max(6.0, Math.min(25.0, rate));
            
            // Financial calculations
            const revenue = loanAmnt * (rate * 0.01) * 3;
            const expectedLoss = risk * loanAmnt * 0.6;
            const operatingCost = loanAmnt * 0.05;
            const profit = revenue - expectedLoss - operatingCost;
            
            // Decision logic
            let decision;
            if (risk < 0.15) decision = 'APPROVE';
            else if (risk < 0.25) decision = 'CONDITIONAL';
            else if (risk < 0.35) decision = 'REVIEW';
            else decision = 'DECLINE';
            
            let riskTier;
            if (risk < 0.10) riskTier = 'Prime';
            else if (risk < 0.15) riskTier = 'Near Prime';
            else if (risk < 0.25) riskTier = 'Subprime';
            else riskTier = 'High Risk';
            
            return {
                apr: Math.round(rate * 100) / 100,
                risk: Math.round(risk * 10000) / 10000,
                profit: Math.round(profit),
                revenue: Math.round(revenue),
                expectedLoss: Math.round(expectedLoss),
                operatingCost: Math.round(operatingCost),
